# Creo los controller
# controllers.py
```python
# -*- coding: utf-8 -*-
from odoo import http
from odoo.http import request

class TallerController(http.Controller):

    # 1. Página Estática: Bienvenida
    @http.route('/taller/bienvenida', auth='public', website=True)
    def index(self, **kw):
        return request.render('taller.bienvenida_page')

    # 2. Página Dinámica: Listado de Vehículos
    @http.route('/taller/vehiculos', auth='public', website=True)
    def list_vehiculos(self, **kw):
        # Obtenemos todos los vehículos de la base de datos
        vehiculos = request.env['taller.vehiculo'].search([])
        
        # Pasamos los datos a la plantilla
        return request.render('taller.lista_vehiculos_page', {
            'vehiculos': vehiculos
        })
```
